Lazy expressions

this

{
    'key': 1,
    'value': {
        'link': 'https://www.example.com/?q=42',
        'list': [1, 2, 3],
        'title': '  Foo bar "1000"  ',
    },
}

this.key

1

this.value.title.strip().upper().replace('"', '')

'FOO BAR 1000'

this.value.list.join(', ')

'1, 2, 3'

this.value.link.urlparse().query.q.cast(int)

42

this.value.title.re(r'\d+').cast(int)

1000

this.value.title.re(r'"(\w+)"')

'1000'

HTML parsing

<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8">
        <title>Sample page</title>
    </head>
    <body>
        <p>Hello world!</p>
    </body>
</html>

p1.select('head > title:text')

[('Sample page', None)]

p1.select('head > title:text', 'body > p:text')

[('Sample page', 'Hello world!')]

p1.select(select('head > title:text').upper())

[('SAMPLE PAGE', None)]

})

            'body': 'Hello world!',
